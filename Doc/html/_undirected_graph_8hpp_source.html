176  void inline _initializeMatrixSystem()
177  {
178  //Calculate counts
179  _nodeCount = static_cast<std::size_t>(std::max(_edges.leftCols(1).maxCoeff(), _edges.rightCols(1).maxCoeff()) + 1);
180  _edgeCount = _edges.rows();
181 
182  // Building Sparse Symmetric Adjacency Metric
183  std::vector<Eigen::Triplet<gSparse::PRECISION>> adjacentList;
184  std::vector<Eigen::Triplet<gSparse::PRECISION>> incidentList;
185  std::vector<Eigen::Triplet<gSparse::PRECISION>> weightList;
186  adjacentList.reserve(_edgeCount * 2);
187  incidentList.reserve(_edgeCount * 2);
188  weightList.reserve(_edgeCount);
189 
190  // Vectorized Zero
191  Eigen::VectorXd degVector = Eigen::VectorXd::Zero(_nodeCount);
192 
193  for (std::size_t i = 0; i != _edgeCount; ++i)
194  {
195  std::size_t r = static_cast<std::size_t>(_edges(i, 0));
196  std::size_t c = static_cast<std::size_t>(_edges(i, 1));
197 
198  //adjacent matrix
199  adjacentList.push_back(Eigen::Triplet<gSparse::PRECISION>(r, c, _weights(i, 0)));
200  adjacentList.push_back(Eigen::Triplet<gSparse::PRECISION>(c, r, _weights(i, 0)));
201 
202  //degree matrix
203  degVector(r) += _weights(i, 0);
204  degVector(c) += _weights(i, 0);
205 
206  //incident matrix
207  if (r != c)
208  {
209  incidentList.push_back(Eigen::Triplet<gSparse::PRECISION>(i, r, 1));
210  incidentList.push_back(Eigen::Triplet<gSparse::PRECISION>(i, c, -1));
211  }
212  //Weight matrix
213  weightList.push_back(Eigen::Triplet<gSparse::PRECISION>(i, i, _weights(i)));
214  }
215  std::cout<<"Adj"<<std::endl;
216  // Create adj matrix
217  _adjMatrix = gSparse::SparsePrecisionMatrix(_nodeCount, _nodeCount);
218  _adjMatrix.setFromTriplets(adjacentList.begin(), adjacentList.end());
219 
220  std::cout<<"Deg"<<std::endl;
221 
222  std::vector<Eigen::Triplet<gSparse::PRECISION>> degreeList;
223  degreeList.reserve(degVector.size());
224  for (int i = 0; i != degVector.size(); ++i)
225  {
226  degreeList.push_back(Eigen::Triplet<gSparse::PRECISION>(i, i, degVector(i)));
227  }
228  //Create degree matrix
229  _degMatrix = gSparse::SparsePrecisionMatrix(_nodeCount, _nodeCount);
230  _degMatrix.setFromTriplets(degreeList.begin(), degreeList.end());
