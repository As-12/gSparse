117  template <typename M>
118  void inline write_csv(const std::string & fileName, const M & matrix)
119  {
120  Eigen::IOFormat CSVFormat(Eigen::StreamPrecision, Eigen::DontAlignCols, _delim, "\n");
121  std::ofstream file(fileName.c_str());
122  if (!file.is_open())
123  {
124  std::stringstream ss;
125  ss << "GraphCSVWriter: File Not Found: " << fileName << std::endl;
126  throw std::runtime_error(ss.str());
127  }
128  file << matrix.format(CSVFormat);
129  file << "\n";
130  file.close();
131  }
