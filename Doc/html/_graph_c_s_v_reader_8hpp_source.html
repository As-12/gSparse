89  template<typename M>
90  void load_csv(const std::string & path, M & matrix)
91  {
92  // Allocate input file stream
93  std::ifstream indata;
94  indata.open(path);
95 
96  // Unable to open the file, throw exceptions
97  if (!indata.is_open())
98  {
99  std::stringstream ss;
100  ss << "GraphCSVReader: File Not Found: " << path << std::endl;
101  throw std::runtime_error(ss.str());
102  }
103 
104  std::string line; // variable to hold each line
105  std::vector<typename M::Scalar> values; // final value
106  std::size_t rows = 0; // row counter
107 
108  // Read each line and add the data into vector . O(rc) complexity. r = rows. c = columns.
109  while (std::getline(indata, line))
110  {
111  std::stringstream lineStream(line);
112  std::string cell;
113  while (std::getline(lineStream, cell, _delim))
114  {
115  values.push_back(static_cast<typename M::Scalar>(std::stod(cell)));
116  }
117  ++rows;
118  }
119  // Close file
120  indata.close();
121  // Map STL vector to Eigen Matrix
122  matrix = Eigen::Map<const Eigen::Matrix<typename M::Scalar, M::RowsAtCompileTime, M::ColsAtCompileTime, Eigen::RowMajor>>(values.data(), rows, values.size() / rows);
