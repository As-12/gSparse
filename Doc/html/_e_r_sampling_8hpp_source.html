71  virtual inline gSparse::Graph GetSparsifiedGraph()
72  {
73  if (_computeInfo == gSparse::NOT_COMPUTED)
74  {
75  throw std::logic_error("SpectralSparsifier by ER: User must run Compute before GetSparsifiedGraph()");
76  }
77  // Build probability distribution
78  std::vector<double> samplingWeights;
79  samplingWeights.reserve(_er.rows());
80 
81  //Generate sampling weights (p) for each edges
82  for (std::size_t i = 0; i != _er.rows(); ++i)
83  {
84  double temp = _er(i, 0) * _graph->GetWeightList()(i) * _c * std::log(_graph->GetNodeCount()) / std::pow(_eps, 2);
85  double prob = 1.0f < temp ? 1.0f : temp;
86  samplingWeights.push_back(prob);
87  }
88 
89  // Build a discrete distribution based on sampling weight
90  std::discrete_distribution<> samplingDistribution(samplingWeights.begin(), samplingWeights.end());
91 
92  // The algorithm samples O(n log n / ep^2) times edges
93  std::size_t samplingCount = static_cast<std::size_t>(
94  std::ceil(_graph->GetNodeCount() * std::log(_graph->GetNodeCount()) / std::pow(_eps, 2)));
95  // Sample edges from a distribution, then add to a hash table
96  std::unordered_map<std::size_t, gSparse::PRECISION> sparsifiedMap;
97  for (std::size_t i = 0; i != samplingCount; ++i)
98  {
99 
100  std::size_t edgeIndex = gSparse::Util::sample(samplingDistribution);
101  sparsifiedMap[edgeIndex] += _graph->GetWeightList()(edgeIndex) / samplingWeights[edgeIndex];
102  }
103 
104  // Build Graph object from sparsified information
105  gSparse::EdgeMatrix resultEdge(sparsifiedMap.size(), 2);
106  gSparse::PrecisionRowMatrix resultWeight(sparsifiedMap.size(), 1);
107  std::size_t row = 0;
108  for (auto x = sparsifiedMap.begin(); x != sparsifiedMap.end(); ++x)
109  {
110  resultEdge(row, 0) = _graph->GetEdgeList()(x->first, 0);
111  resultEdge(row, 1) = _graph->GetEdgeList()(x->first, 1);
112  resultWeight(row, 0) = x->second;
113  ++row;
114  }
115  //return nullptr;
116  return std::make_shared<gSparse::UndirectedGraph>(resultEdge, resultWeight);
117  }
