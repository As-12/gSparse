91  virtual inline gSparse::Graph GetSparsifiedGraph()
92  {
93  if (_computeInfo == gSparse::NOT_COMPUTED)
94  {
95  throw std::logic_error("SpectralSparsifier by ER: User must run Compute before GetSparsifiedGraph()");
96  }
97  // Build probability distribution
98  std::vector<double> samplingWeights;
99  samplingWeights.reserve(_er.rows());
100 
101  //Generate sampling weights (p) for each edges
102  for (std::size_t i = 0; i != _er.rows(); ++i)
103  {
104  double temp = _er(i, 0) * _graph->GetWeightList()(i) * _c * std::log(_graph->GetNodeCount()) / std::pow(_eps, 2);
105  double prob = 1.0f < temp ? 1.0f : temp;
106  samplingWeights.push_back(prob);
107  }
108 
109  // Build a discrete distribution based on sampling weight
110  std::discrete_distribution<> samplingDistribution(samplingWeights.begin(), samplingWeights.end());
111 
112  // The algorithm samples O(n log n / ep^2) times edges
113  std::size_t samplingCount = static_cast<std::size_t>(
114  std::ceil(_graph->GetNodeCount() * std::log(_graph->GetNodeCount()) / std::pow(_eps, 2)));
115  // Sample edges from a distribution, then add to a hash table
116  std::unordered_map<std::size_t, gSparse::PRECISION> sparsifiedMap;
117  for (std::size_t i = 0; i != samplingCount; ++i)
118  {
119 
120  std::size_t edgeIndex = gSparse::Util::sample(samplingDistribution);
121  sparsifiedMap[edgeIndex] += _graph->GetWeightList()(edgeIndex) / samplingWeights[edgeIndex];
122  }
123 
124  // Build Graph object from sparsified information
125  gSparse::EdgeMatrix resultEdge(sparsifiedMap.size(), 2);
126  gSparse::PrecisionRowMatrix resultWeight(sparsifiedMap.size(), 1);
127  std::size_t row = 0;
128  for (auto x = sparsifiedMap.begin(); x != sparsifiedMap.end(); ++x)
129  {
130  resultEdge(row, 0) = _graph->GetEdgeList()(x->first, 0);
131  resultEdge(row, 1) = _graph->GetEdgeList()(x->first, 1);
132  resultWeight(row, 0) = x->second;
133  ++row;
134  }
135  return std::make_shared<gSparse::UndirectedGraph>(resultEdge, resultWeight);
136  }
