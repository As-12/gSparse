21  const gSparse::Graph & graph,
22  double eps = 1.0f,
23  double JLTol = 0.5,
24  int maxIter = 300
25  )
26  {
27  er = gSparse::PrecisionRowMatrix::Zero(graph->GetEdgeCount(), 1);
28 
29  std::size_t scale = static_cast<size_t>(
30  std::ceil(
31  std::log2(
32  static_cast<double>(graph->GetIncidentMatrix().cols()) / eps)));
33 
34  for (int i = 1; i != scale + 1; ++i)
35  {
36  Eigen::VectorXd x;
37  gSparse::PrecisionMatrix Q =
38  gSparse::Util::randomProjectionMatrix(1,
39  graph->GetIncidentMatrix().rows(),
40  static_cast<double>(scale),
41  JLTol);
42 
43  gSparse::PrecisionMatrix Y = (Q * graph->GetWeightMatrix().cwiseSqrt() * graph->GetIncidentMatrix());
44 
45  // solve Linear system with 300 max iteration
46  Eigen::ConjugateGradient<gSparse::SparsePrecisionMatrix, Eigen::Lower | Eigen::Upper > cg;
47  cg.setMaxIterations(maxIter);
48  x = cg.compute(graph->GetLaplacianMatrix()).solve(Y.transpose());
49 
50  if (cg.info() != Eigen::Success)
51  {
52  // Does not converge this iteration. Keeps going.
53  continue;
54  }
55  for (std::size_t j = 0; j != graph->GetEdgeCount(); ++j)
56  {
57  er(j) += pow(std::abs(x(graph->GetEdgeList()(j, 0)) - x(graph->GetEdgeList()(j, 1))), 2.0f);
58  }
59  }
60  // Non finite element goes to zero
61  er = er.unaryExpr([](double v) { return std::isfinite(v)? v : 0.0; });
62  if (er.rows() != graph->GetEdgeCount())
63  return gSparse::NOT_CONVERGING;
64  return gSparse::SUCCESSFUL;
65  }
